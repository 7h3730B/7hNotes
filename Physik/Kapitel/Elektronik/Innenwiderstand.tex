\subsection{Innenwiderstand von Spannungswandlern}
$I_k$ ist bei U = 0 und $U_0$ ist bei I = 0
\begin{itemize}
    \item $R_g = R_i + R_a$
    \item $U_0 = U_i + U_a$
    \item $U_a = U_0 - U_i$
    \item $U_a = U_0 - R_i * I$
\end{itemize}
Wenn $R_a$ = 0 ist, dann entsteht ein Kurzschluss 
\begin{itemize}
    \item => $U_a = 0$
    \item => $0 = U_0 - R_i * I$
    \item => $R_i = \frac{U_0}{I}$
\end{itemize}
